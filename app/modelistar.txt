class Order(models.Model):
    author = models.ForeignKey(settings.AUTH_USER_MODEL, on_delete=models.CASCADE)
    place_start = models.CharField(max_length=200)
    place_end = models.CharField(max_length=200)
    tariff = models.TextField()
    numder = models.CharField(max_length=10)
    date = models.DateTimeField(default=timezone.now)
    other = models.TextField(default='в обработке')
    published_date = models.DateTimeField(blank=True, null=True)

    def publish(self):
        self.published_date = timezone.now()
        self.save()

    def __str__(self):
        return self.place_end
